2. fsm_no_hash.rb

1: file = 'README_sample.org'
2: state = :head_search
3: File.readlines(file)[12..15].each do |line|
4:   p [state, line]
5:   state = :block_search if line.match(/\*\* /)
6:   state = :head_search if line.match(/^-/)
7: end
